LeetCode-1044-最长重复子串

2021-12-23 2 分钟阅读 LeetCodeleetcode困难

题目

给你一个字符串 s ，考虑其所有 重复子串 ：即，s 的连续子串，在 s 中出现 2 次或更多次。这些出现之间可能存在重叠。

返回 任意一个 可能具有最长长度的重复子串。如果 s 不含重复子串，那么答案为 "" 。

示例

1输入：s = "banana"
2输出："ana"

1输入：s = "abcd"
2输出：""

解答

  1class SuffixArray {
  2public:
  3    using size_type = unsigned;
  4    using pointer = size_type*;
  5    using const_pointer = const size_type*;
  6    const_pointer sa, rk, ht;
  7
  8private:
  9    std::unique_ptr<size_type[]> data;
 10
 11private:
 12    template<typename S>
 13    inline static bool substring_equal(const S& s, size_type p1, size_type p2, size_type len) {
 14        for (size_type i = 0;i < len;++i)
 15            if (s[p1 + i] != s[p2 + i])
 16                return false;
 17        return true;
 18    }
 19
 20    template<typename S>
 21    inline static void induced_sort(
 22        const S& s,
 23        pointer sa,
 24        bool* type,
 25        pointer pos,
 26        pointer lbuk,
 27        pointer sbuk,
 28        size_type n,
 29        size_type m,
 30        size_type n0) {
 31        std::fill_n(sa, n, 0);
 32        lbuk[0] = 0;
 33        for (size_type i = 1;i < m;++i)
 34            lbuk[i] = sbuk[i - 1];
 35        for (size_type i = n0;i-- > 0;)
 36            sa[--sbuk[s[pos[i]]]] = pos[i];
 37        sbuk[m - 1] = n;
 38        for (size_type i = 1;i < m;++i)
 39            sbuk[i - 1] = lbuk[i];
 40        sa[lbuk[s[n - 1]]++] = n - 1;
 41        for (size_type i = 0;i < n;++i)
 42            if (sa[i] > 0 && !type[sa[i] - 1])
 43                sa[lbuk[s[sa[i] - 1]]++] = sa[i] - 1;
 44        lbuk[0] = 0;
 45        for (size_type i = 1;i < m;++i)
 46            lbuk[i] = sbuk[i - 1];
 47        for (size_type i = n;i-- > 0;)
 48            if (sa[i] > 0 && type[sa[i] - 1])
 49                sa[--sbuk[s[sa[i] - 1]]] = sa[i] - 1;
 50    }
 51
 52    template<typename S>
 53    static void sais(
 54        const S& s,
 55        pointer sa,
 56        bool* type,
 57        pointer len,
 58        pointer pos,
 59        pointer lbuk,
 60        pointer sbuk,
 61        size_type n,
 62        size_type m) {
 63        type[n - 1] = false;
 64        for (size_type i = n - 1;i-- > 0;)
 65            type[i] = s[i] != s[i + 1] ? s[i] < s[i + 1] : type[i + 1];
 66        size_type n0 = 0;
 67        for (size_type i = 1;i < n;++i)
 68            if (!type[i - 1] && type[i])
 69                pos[n0++] = i;
 70        std::fill_n(len, n, 0);
 71        for (size_type p = n - 1, i = n0;i-- > 0;p = pos[i])
 72            len[pos[i]] = p - pos[i] + 1;
 73        std::fill_n(sbuk, m, 0);
 74        for (size_type i = 0;i < n;++i)
 75            ++sbuk[s[i]];
 76        for (size_type i = 1;i < m;++i)
 77            sbuk[i] += sbuk[i - 1];
 78        induced_sort(s, sa, type, pos, lbuk, sbuk, n, m, n0);
 79        sbuk[m - 1] = n;
 80        for (size_type i = 1;i < m;++i)
 81            sbuk[i - 1] = lbuk[i];
 82        size_type m0 = -1;
 83        size_type ppos = -1, plen = 0;
 84        for (size_type i = 0;i < n;++i) {
 85            if (len[sa[i]] == 0) continue;
 86            if (len[sa[i]] != plen || !substring_equal(s, sa[i], ppos, plen)) ++m0;
 87            plen = len[sa[i]];
 88            len[sa[i]] = m0;
 89            ppos = sa[i];
 90        }
 91        pointer s0 = sa;
 92        pointer sa0 = sa + n0;
 93        for (size_type i = 0;i < n0;++i)
 94            s0[i] = len[pos[i]];
 95        if (++m0 < n0)
 96            sais(s0, sa0, type + n, len, pos + n0, lbuk, lbuk + n0, n0, m0);
 97        else for (size_type i = 0;i < n0;++i)
 98            sa0[s0[i]] = i;
 99        for (size_type i = 0;i < n0;++i)
100            pos[i + n0] = pos[sa0[i]];
101        induced_sort(s, sa, type, pos + n0, lbuk, sbuk, n, m, n0);
102    }
103
104public:
105    template<typename S>
106    SuffixArray(const S& s, size_type n, size_type m)
107        : data(std::make_unique<size_type[]>(3 * n)) {
108        const auto type = std::make_unique<bool[]>(2 * n);
109        const auto lbuk = std::make_unique<size_type[]>(std::max(n, m));
110        const auto sbuk = std::make_unique<size_type[]>(m);
111        pointer sa = data.get(), rk = sa + n, ht = rk + n;
112        sais(s, sa, type.get(), rk, ht, lbuk.get(), sbuk.get(), n, m);
113        for (size_type i = 0;i < n;++i)
114            rk[sa[i]] = i;
115        for (size_type k = 0, i = 0;i < n;++i) {
116            if (rk[i] == 0) continue;
117            if (k > 0) --k;
118            for (size_type j = sa[rk[i] - 1], l = n - std::max(i, j);k < l;++k)
119                if (s[i + k] != s[j + k]) break;
120            ht[rk[i]] = k;
121        }
122        this->sa = sa;
123        this->rk = rk;
124        this->ht = ht;
125    }
126
127    inline size_type suffix(size_type p) const {
128        return sa[p];
129    }
130
131    inline size_type rank(size_type p) const {
132        return rk[p];
133    }
134
135    inline size_type height(size_type p) const {
136        return ht[p];
137    }
138};

140class Solution {
141public:
142    string longestDupSubstring(string s) {
143        const int n = s.size();
144        SuffixArray sa(s, n, 128);
145        int len = 0, pos = -1;
146        for (int i = 1;i < n;++i) {
147            if (sa.ht[i] > len) {
148                len = sa.ht[i];
149                pos = sa.sa[i];
150            }
151        }
152        return pos == -1 ? "" : s.substr(pos, len);
153    }
154};
