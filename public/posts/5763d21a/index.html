从0到1实现完整的微服务框架-服务注册、发现、配置中心

2022-03-28 5 分钟阅读 从0到1实现完整的微服务框架微服务gRPC配置中心服务发现服务注册

当某一个服务要以集群的形式进行部署，这时候就要用到服务注册和服务发现。主要介绍使用consul进行服务发现、服务注册以及配置中心。

​ 假如这个产品已经在线上运行，有一天运营想搞一场促销活动，那么我们相对应的【用户服务】可能就要新开启三个微服务实例来支撑这场促销活动。而与此同时，作为苦逼程序员的你就只有手动去 API gateway 中添加新增的这三个微服务实例的 ip 与port ，一个真正在线的微服务系统可能有成百上千微服务，难道也要一个一个去手动添加吗？有没有让系统自动去实现这些操作的方法呢？答案当然是有的。 ​ 当我们新添加一个微服务实例的时候，微服务就会将自己的 ip 与 port 发送到注册中心，在注册中心里面记录起来。当 API gateway 需要访问某些微服务的时候，就会去注册中心取到相应的 ip 与 port。从而实现自动化操作。

将grpc的服务注册到consul

添加consul的配置shop\service\user_srv\config\consul_info.go

 1type ServerConfig struct{
 2    Host string `mapstructure:"host"` // 服务启动的host
 3	Port int    `mapstructure:"port"` // 服务启动的port
 4	Name string `mapstructure:"name"` // 服务的名称
 5    Consul 	Consul `mapstructure:"consul"` // consul的配置
 6}
 7
 8type Consul struct {
 9    Host string `mapstructure:"host"` // consul的host
10	Port int    `mapstructure:"port"` // consul的port
11}

这里问什么要在severConfig中添加一个name，是因为每一个服务都有一个自己的服务名称，在http客户端之后找服务的时候就找服务名下的一个服务就行。

在main中

 1import(
 2    "github.com/hashicorp/consul/api
 3)
 4
 5func main(){

13	if err != nil {

15	}

18	check := api.AgentServiceCheck{
19		GRPC: fmt.Sprintf("%s:%d",
20			global.ServerConfig.Host, // 这里就是你要检测那个服务的host和port
21			global.ServerConfig.Port,
22		),
23		Timeout:                        "3s",
24		Interval:                       "5s",
25		DeregisterCriticalServiceAfter: "100s", // 健康检查失败超过100s之后就会删除这个服务
26	}
27    // 一个服务中可以
28    var serviceID uuid.UUID
29	for {
30		serviceID, err = uuid.NewRandom()
31		if err == nil {
32			break
33		}
34	}
35
36	// consul 做健康检查的ip
37	serviceRegistration := api.AgentServiceRegistration{
38		ID:      serviceID.String(),
39		Name:    global.RemoteConfig.ServiceInfo.Name,
40		Port:    global.RemoteConfig.ServiceInfo.Port,
41		Address: global.RemoteConfig.ServiceInfo.Host,
42	}
43

用consul做注册中心

在consul是有存储kv的数据库，基于这个我们就可以将配置保存在consul中。具体的实现也很简单，之前我们用viper读本地的配置文件，而viper的强大在于他也能读远程的配置文件。

这时候，我们本地只需要保存consul的配置就行，首先加载consul的位置，之后从consul中读取配置文件，就可以。

shop\api\user-api\initialize\config.go这是api的加载配置文件，同样，srv也是一样的。

 1package initialize
 2
 3import (
 4	"fmt"
 5
 6	"github.com/fsnotify/fsnotify"
 7	"github.com/spf13/viper"
 8	_ "github.com/spf13/viper/remote"
 9	"go.uber.org/zap"
10
11	"github.com/jimyag/shop/api/user/global"
12)
13
14func getEnvBool(env string) bool {
15	viper.AutomaticEnv()
16	return viper.GetBool(env)
17}
18
19// LoadConsulConfigInfo 加载本地的 consul 文件
20func LoadConsulConfigInfo() {
21	configFilePath := "consul-info.yaml"
22
23	v := viper.New()
24	v.SetConfigFile(configFilePath)
25	if err := v.ReadInConfig(); err != nil {
26		global.Logger.Fatal("加载配置文件失败.....",
27			zap.Error(err),
28			zap.String("path", configFilePath),
29		)
30	}
31
32	global.Logger.Info("配置加载成功....",
33		zap.String("path", configFilePath),
34	)
35	if err := v.Unmarshal(&global.ConsulCenterInfo); err != nil {
36		global.Logger.Fatal("解析配置文件失败....",
37			zap.Error(err),
38			zap.String("path", configFilePath),
39		)
40	}
41	global.Logger.Info("成功加载配置文件",
42		zap.String("path", configFilePath),
43		zap.Any("content", global.ConsulCenterInfo),
44	)
45	v.WatchConfig()
46	v.OnConfigChange(func(in fsnotify.Event) {
47		global.Logger.Info("配置文件产生变化....",
48			zap.String("name", in.String()),
49			zap.String("path", configFilePath),
50		)
51
52		if err := v.ReadInConfig(); err != nil {
53			global.Logger.Fatal("修改的配置文件字段出错",
54				zap.String("field", in.String()),
55				zap.Error(err),
56			)
57		}
58		if err := v.Unmarshal(&global.ConsulCenterInfo); err != nil {
59			global.Logger.Fatal("解析配置文件出错",
60				zap.String("field", in.String()),
61				zap.Error(err),
62			)
63		}
64		global.Logger.Info("配置文件内容", zap.Any("config", global.ConsulCenterInfo))
65	})
66}
67
68func LoadRemoteConfig() {
69	remoteViper := viper.New()
70	path := global.ConsulCenterInfo.ReleasePath
71	if debug := getEnvBool(global.ConsulCenterInfo.EnvName); debug {
72		path = global.ConsulCenterInfo.DebugPath
73	}
74	remoteViper.SetConfigType(global.ConsulCenterInfo.FileType)
75	err := remoteViper.AddRemoteProvider(global.ConsulCenterInfo.Type,
76		fmt.Sprintf("%s:%d", global.ConsulCenterInfo.Host,
77			global.ConsulCenterInfo.Port,
78		),
79		path,
80	)
81	if err != nil {
82		global.Logger.Fatal("添加配置文件失败", zap.Error(err))
83		return
84	}
85
86	err = remoteViper.ReadRemoteConfig()
87	if err != nil {
88		global.Logger.Fatal("读取远端配置文件失败", zap.Error(err))
89	}
90	err = remoteViper.Unmarshal(&global.ServerConfig)
91	if err != nil {
92		global.Logger.Fatal("解析远端配置文件失败", zap.Error(err))
93	}
94	global.Logger.Info("成功加载远端配置文件....", zap.Any("config", global.ServerConfig))
95}
